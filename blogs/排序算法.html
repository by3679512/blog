    排序算法

    # 选择排序

    在循环中，选择未排序区间中最小的值，放到已排序区间的最后，直到未排序区间遍历结束。 设数组长度为n，算法步骤如下：

    • 初始时，所有元素均未排序，此时未排序区间为[0,n-1]
    • 在未排序区间选择最小元素，与索引为0的元素交换，完成后，完成第一个元素的排序
    • 重复第二步骤直到未排序区间全部遍历完毕
    function selectionSort(nums) {
    let n = nums.length;
    for (let i = 0; i < n-1; i++) {
        let k = i
        for (let j = i; j < n;j++){
            if(nums[j]<nums[k]){
                k = j;
            }
        }
        [nums[i], nums[k]] = [nums[k], nums[i]]   
    } 
}

    时间复杂度：O(n2n^2n​2​​)

    # 插入排序

    在循环中，将未排序区间的第一个元素插入到已排序区间的合适位置，直到未排序区间遍历结束。 设数组长度为n，算法步骤如下：

    • 初始时，所有元素均未排序，此时已排序区间为[0,0]
    • 将未排序区间的第一个元素插入到已排序区间的合适位置，完成后，已排序区间为[0,1]
    • 重复第二步骤直到未排序区间全部遍历完毕
    • 仅剩的一个元素必定是最小元素，无需排序，因此数组排序完成

    补充

    • 由于已排序区间的长度逐渐增加，因此可以使用二分查找法寻找插入位置，将时间复杂度降低至O(nlognnlognnlogn)
    • 由于数组元素的移动操作比较耗时，因此可以使用哨兵，将数组元素的移动操作降低至O(1)
    function insertionSort(nums) {
    let n = nums.length;
    for (let i = 1; i < n; i++) {
        let j = i;
        while(j-1>=0 && nums[j]<nums[j-1]){
            [nums[j], nums[j-1]] = [nums[j-1], nums[j]];
            j--;
        }
    }
}

    时间复杂度：O(n2n^2n​2​​) 

    function insertionSort(nums) {
    let n = nums.length;
    for (let i = 1; i < n; i++) {
        let left = 0;
        let right = i-1;
        let temp = nums[i];
        while(left<=right){
            let mid = Math.floor((left+right)/2);
            if(nums[mid]>temp){
                right = mid-1;
            }else{
                left = mid+1;
            }
        }
        for (let j = i-1; j >= left; j--) {
            nums[j+1] = nums[j];
        }
        nums[left] = temp;
    }
}

    时间复杂度：O(nlognnlognnlogn)

    # 归并排序

    在循环中，将数组分为两个区间，分别对两个区间进行排序，最后将两个有序区间合并为一个有序区间。 设数组长度为n，算法步骤如下：

    • 初始时，将数组分为 n 个区间，每个区间只有一个元素，此时每个区间均为有序区间
    • 将两个有序区间合并为一个有序区间，此时有序区间的长度为2
    • 重复第二步骤直到有序区间的长度等于数组长度
    function mergeSort(nums) {
    function merge(nums, left, mid, right) {
        let temp = [];
        let i = left;
        let j = mid+1;
        let k = 0;
        while(i<=mid && j<=right){
            if(nums[i]<=nums[j]){
                temp[k++] = nums[i++];
            }else{
                temp[k++] = nums[j++];
            }
        }
        while(i<=mid){
            temp[k++] = nums[i++];
        }
        while(j<=right){
            temp[k++] = nums[j++];
        }
        for (let i = 0; i < temp.length; i++) {
            nums[left+i] = temp[i];
        }
    }
    function sort(nums, left, right) {
        if(left<right){
            let mid = Math.floor((left+right)/2);
            sort(nums, left, mid);
            sort(nums, mid+1, right);
            merge(nums, left, mid, right);
        }
    }
    sort(nums, 0, nums.length-1);
}

    时间复杂度：O( nlognnlognnlogn )

    # 冒泡排序

    • 首先，对n个元素执行“冒泡”，将数组的最大元素交换至正确位置
    • 接下来，对剩余n-1个元素执行“冒泡”，将第二大元素交换至正确位置
    • 以此类推，经过n-1轮“冒泡”后，前n-1大的元素都被交换至正确位置
    • 仅剩的一个元素必定是最小元素，无需排序，因此数组排序完成
    function bubbleSort(nums) {
    for (let i = nums.length-1; i > 0;i--) {
        for (let j = 0; j < i;j++){
            if(nums[j]>nums[j+1]){
                [nums[j], nums[j+1]] = [nums[j+1], nums[j]]
            }
        }
    }
}

    时间复杂度：O( n2n^2n​2​​ )

    # 希尔排序

    • 选择一个增量序列t1，t2，…，tk，其中ti>tj，tk=1；
    • 按增量序列个数k，对序列进行k 趟排序；
    • 每趟排序，根据对应的增量ti，将待排序列分割成若干长度为m 的子序列，分别对各子表进行直接插入排序。仅增量因子为1 时，整个序列作为一个表来处理，表长度即为整个序列的长度。
    • 希尔排序是不稳定的排序算法，虽然时间复杂度达不到O(nlogn)，但是比O(n^2)要好
    function shellSort(nums) {
    let n = nums.length;
    for (let gap = Math.floor(n/2); gap > 0; gap = Math.floor(gap/2)) {
        for (let i = gap; i < n; i++) {
            let j = i;
            while(j-gap>=0 && nums[j]<nums[j-gap]){
                [nums[j], nums[j-gap]] = [nums[j-gap], nums[j]]
                j -= gap;
            }
        }
    }
}

    时间复杂度：O( n2n^2n​2​​ )

    # 快速排序

    • 从数列中挑出一个元素，称为 “基准”（pivot）；
    • 重新排序数列，所有元素比基准值小的摆放在基准前面，所有元素比基准值大的摆在基准的后面（相同的数可以到任一边）。在这个分割之后，该基准是它的最后位置。这个称为分割（partition）操作；
    • 递归地（recursive）把小于基准值元素的子数列和大于基准值元素的子数列排序。
    • 递归到最后，数组排序完成
    • 快速排序是不稳定的排序算法，时间复杂度达到O(nlogn)
    • 快速排序的优化：三数取中法，随机法，插入排序法
    • 三数取中法：取left，mid，right三个数中的中间值作为基准值
    • 随机法：随机选择一个数作为基准值
    • 插入排序法：当数组长度小于某个值时，使用插入排序法
    • 三数取中法和随机法的效率相差不大，但是三数取中法的效率更稳定
    • 插入排序法的效率最高，但是当数组长度较大时，效率会很低
    function quickSort(nums) {
    function partition(nums, left, right) {
        let pivot = nums[left];
        while(left<right){
            while(left<right && nums[right]>=pivot){
                right--;
            }
            nums[left] = nums[right];
            while(left<right && nums[left]<=pivot){
                left++;
            }
            nums[right] = nums[left];
        }
        nums[left] = pivot;
        return left;
    }
    function sort(nums, left, right) {
        if(left<right){
            let pivotIndex = partition(nums, left, right);
            sort(nums, left, pivotIndex-1);
            sort(nums, pivotIndex+1, right);
        }
    }
    sort(nums, 0, nums.length-1);
}

    时间复杂度：O( nlognnlognnlogn )

    # 堆排序

    • 将待排序序列构造成一个大顶堆
    • 此时，整个序列的最大值就是堆顶的根节点
    • 将其与末尾元素进行交换，此时末尾就为最大值
    • 然后将剩余n-1个元素重新构造成一个堆，这样会得到n个元素的次小值
    • 如此反复执行，便能得到一个有序序列了
    • 堆排序是不稳定的排序算法，时间复杂度达到O(nlogn)
    • 堆排序的优化：原地堆排序
    • 原地堆排序：将堆排序的空间复杂度降为O(1)
    • 原地堆排序的实现：将堆排序的数组下标从0开始，这样就可以将堆排序的空间复杂度降为O(1)
    function heapSort(nums) {
    function heapify(nums, i, size) {
        let left = 2*i+1;
        let right = 2*i+2;
        let max = i;
        if(left<size && nums[left]>nums[max]){
            max = left;
        }
        if(right<size && nums[right]>nums[max]){
            max = right;
        }
        if(max!=i){
            [nums[i], nums[max]] = [nums[max], nums[i]];
            heapify(nums, max, size);
        }
    }
    function buildHeap(nums, size) {
        for (let i = Math.floor(size/2); i >= 0; i--) {
            heapify(nums, i, size);
        }
    }
    let size = nums.length;
    buildHeap(nums, size);
    for (let i = size-1; i > 0; i--) {
        [nums[0], nums[i]] = [nums[i], nums[0]];
        size--;
        heapify(nums, 0, size);
    }
}

    时间复杂度：O( nlognnlognnlogn )

    # 桶排序

    • 设置固定数量的空桶
    • 把数据放到对应的桶中
    • 对每个不为空的桶中数据进行排序
    • 拼接不为空的桶中数据，得到结果
    • 桶排序是稳定的排序算法，时间复杂度达到O(n)

    # 桶排序的优化

    • 桶排序的优化：桶的数量，桶的大小
    • 桶的数量：桶的数量越多，排序的效率越高，但是空间复杂度也越高
    • 桶的大小：桶的大小越大，排序的效率越高，但是空间复杂度也越高
    function bucketSort(nums) {
    let max = Math.max(...nums);
    let min = Math.min(...nums);
    let bucketSize = 5;
    let bucketCount = Math.floor((max-min)/bucketSize)+1;
    let buckets = new Array(bucketCount);
    for (let i = 0; i < bucketCount; i++) {
        buckets[i] = [];
    }
    for (let i = 0; i < nums.length; i++) {
        let index = Math.floor((nums[i]-min)/bucketSize);
        buckets[index].push(nums[i]);
    }
    let res = [];
    for (let i = 0; i < buckets.length; i++) {
        if(buckets[i].length>0){
            insertionSort(buckets[i]);
            res.push(...buckets[i]);
        }
    }
    return res;
}

    时间复杂度：O( n2n^2n​2​​ )

    # 基数排序

    • 基数排序是一种非比较型整数排序算法，其原理是将整数按位数切割成不同的数字，然后按每个位数分别比较
    • 基数排序是稳定的排序算法，时间复杂度达到O(n)
    • 基数排序的实现：LSD(Least significant digital)和MSD(Most significant digital)
    • LSD：从最低位开始，依次进行排序
    • MSD：从最高位开始，依次进行排序
    • LSD的实现：用到了桶排序
    • MSD的实现：用到了递归
    function radixSort(nums) {
    let max = Math.max(...nums);
    let maxDigit = 0;
    while(max>0){
        max = Math.floor(max/10);
        maxDigit++;
    }
    let buckets = new Array(10);
    for (let i = 0; i < 10; i++) {
        buckets[i] = [];
    }
    for (let i = 0; i < maxDigit; i++) {
        for (let j = 0; j < nums.length; j++) {
            let index = Math.floor(nums[j]/Math.pow(10,i))%10;
            buckets[index].push(nums[j]);
        }
        let k = 0;
        for (let j = 0; j < 10; j++) {
            if(buckets[j].length>0){
                for (let l = 0; l < buckets[j].length; l++) {
                    nums[k++] = buckets[j][l];
                }
                buckets[j] = [];
            }
        }
    }
}

    时间复杂度：O( n∗kn*kn∗k )

    # 计数排序

    • 计数排序是一种非比较型整数排序算法，其原理是统计每个整数在序列中出现的次数，进而推导出每个整数在有序序列中的索引
    • 计数排序是稳定的排序算法，时间复杂度达到O(n)
    • 计数排序的实现：用到了桶排序
    function countingSort(nums) {
    let max = Math.max(...nums);
    let min = Math.min(...nums);
    let buckets = new Array(max-min+1).fill(0);
    for (let i = 0; i < nums.length; i++) {
        buckets[nums[i]-min]++;
    }
    let k = 0;
    for (let i = 0; i < buckets.length; i++) {
        while(buckets[i]>0){
            nums[k++] = i+min;
            buckets[i]--;
        }
    }
}

    时间复杂度：O( n+kn+kn+k )
